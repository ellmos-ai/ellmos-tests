# Stand: 2026-02-18 | Bezug: HQ1 (Dateizuordnung prüfen)

# Diese Batterie ist idempotent - kann jederzeit zur Verifikation genutzt werden.
# Hilfreich nach Bulk-Updates, Migrationen oder neuen Einträgen.

# ============================================================
# SKILLS TABELLE
# ============================================================

DT001 | skills | Extension-Skills (C: Pfad) = USER=0 (außer 4 Template-Extensions)
  SQL: SELECT name, path, dist_type FROM skills
       WHERE path LIKE 'C:%' AND dist_type != 0
       AND name NOT IN ('FavExtract','MetaWiki','SQLiteViewer','UniversalCompiler')
  Erwartet: Leeres Ergebnis

DT002 | skills | Template-Extensions = TEMPLATE=1
  SQL: SELECT name, dist_type FROM skills
       WHERE name IN ('FavExtract','MetaWiki','SQLiteViewer','UniversalCompiler')
       AND dist_type != 1
  Erwartet: Leeres Ergebnis

DT003 | skills | Foerderplaner = USER=0
  SQL: SELECT name, dist_type FROM skills
       WHERE path LIKE 'agents/_experts/foerderplaner%' AND dist_type != 0
  Erwartet: Leeres Ergebnis

DT004 | skills | Worksheet-Generator = USER=0
  SQL: SELECT name, dist_type FROM skills
       WHERE path LIKE 'agents/_experts/worksheet_generator%' AND dist_type != 0
  Erwartet: Leeres Ergebnis

DT005 | skills | recludos.md = USER=0
  SQL: SELECT name, dist_type FROM skills WHERE path = 'skills/_os/recludos.md'
  Erwartet: dist_type = 0

DT006 | skills | copilot.md (lowercase) = USER=0
  SQL: SELECT name, dist_type FROM skills WHERE path = 'skills/_templates/copilot.md'
  Erwartet: dist_type = 0

DT007 | skills | skills/_protocols/ = CORE=2
  SQL: SELECT COUNT(*) FROM skills WHERE path LIKE 'skills/_protocols/%' AND dist_type != 2
  Erwartet: 0

DT008 | skills | skills/_services/ = CORE=2
  SQL: SELECT COUNT(*) FROM skills WHERE path LIKE 'skills/_services/%' AND dist_type != 2
  Erwartet: 0

DT009 | skills | Agents (außer USER-Ausnahmen) = TEMPLATE=1
  SQL: SELECT name, path, dist_type FROM skills
       WHERE path LIKE 'agents/%'
       AND path NOT LIKE 'agents/_experts/foerderplaner%'
       AND path NOT LIKE 'agents/_experts/worksheet_generator%'
       AND dist_type NOT IN (1, 2)
  Erwartet: Leeres Ergebnis (oder maximal wenige unklare Grenzfälle)
